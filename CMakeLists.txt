cmake_minimum_required(VERSION 3.21)
project(accurate_kosinski LANGUAGES C)
include_directories(lib)

option(DEBUG "Enable Debug information" OFF)
option(COMPARE "Build kcompare for tests" OFF)
option(ONLYLIB "Only build library" OFF)


if (DEBUG)
	add_compile_options("-DDEBUG")
	set(CMAKE_BUILD_TYPE "Debug" CACHE STRING "Choose Release or Debug" FORCE)
else()
	if(NOT CMAKE_BUILD_TYPE)
		set(CMAKE_BUILD_TYPE "Release" CACHE STRING "Choose Release or Debug" FORCE)
	endif()
endif()

file(GLOB_RECURSE KOSINSKI_SOURCE lib/*.c)
file(GLOB_RECURSE KOSINSKI_HEADER lib/*.h)

add_library(kosinski SHARED
	${KOSINSKI_SOURCE}
	${KOSINSKI_HEADER}
)

if(NOT ONLYLIB)
	add_executable(kcompress
		"main_compress.c"
		"load_file_to_buffer.c"
	)
	target_link_libraries(kcompress kosinski)

	add_executable(kcompress_asm
		"main_compress_asm.c"
		"load_file_to_buffer.c"
	)
	target_link_libraries(kcompress_asm kosinski)

	add_executable(kdecompress
		"main_decompress.c"
		"load_file_to_buffer.c"
	)
	target_link_libraries(kdecompress kosinski)

	add_executable(kmdecompress
		"main_moduled_decompress.c"
		"load_file_to_buffer.c"
	)
	target_link_libraries(kmdecompress kosinski)

	add_executable(kmcompress
		"main_moduled_compress.c"
		"load_file_to_buffer.c"
	)
	target_link_libraries(kmcompress kosinski)

	set_target_properties(kcompress kcompress_asm kdecompress kmcompress kmdecompress
		PROPERTIES
			C_STANDARD 99
			C_EXTENSIONS OFF
	)
	if(COMPARE)
		add_executable(kmcompare
			"main_moduled_compare.c"
			"load_file_to_buffer.c"
		)
		target_link_libraries(kmcompare kosinski)

		add_executable(kcompare
			"main_compare.c"
			"load_file_to_buffer.c"
		)
		target_link_libraries(kcompare kosinski)

		set_target_properties(kcompare kmcompare 
			PROPERTIES
				C_STANDARD 99
				C_EXTENSIONS OFF
		)
	endif()
endif()
